    \begin{axis}
      [
        legend pos=north west,
        legend style={font=\footnotesize\itshape,cells={anchor=west}},
        ybar=10pt,
        bar width=17pt,
        cycle list name=TUM-corporate,
        /pgf/bar shift=0pt,scale only axis,
        height=0.3\textwidth,
        width=0.35\textwidth,
        ylabel style={%
            at={(axis description cs:0,1)},
            anchor=south,
            rotate=-90,
          },
        ylabel=$\mu$,
        enlarge x limits=0.15,
        ymin=0,
        xtick=\empty,
        legend style={draw=none, /tikz/every even column/.append style={column sep=5pt}},
        legend image code/.code={%
            \draw[##1] (0cm,-0.1cm) rectangle (0.3cm,0.1cm);
          }
      ]

      \pgfplotsinvokeforeach{0,...,#2}{
        \addplot table [ x expr=\coordindex, select row=##1, y=Mean] {\tempdata};
        \addlegendentry {\pgfplotstablegetelem{##1}{Method}\of\tempdata \pgfplotsretval}
      }

    \end{axis}
  \end{tikzpicture}
}
